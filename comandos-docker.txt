# Checar a porta (Linux):
ss -tulpn | grep 5432

# Checar a porta (Windows):
netstat -aof | findstr 5432

# Rodar container Postgre na versão 16.3
docker run --name librarydb -e POSTGRES_PASSWORD=postgres -e POSTGRES_USER=postgres -e POSTGRES_DB=library -p 5432:5432 -d --network library-network postgres:16.3

# Rodar Pgadmin 4
docker run --name pgadmin4 -e PGADMIN_DEFAULT_EMAIL=admin@admin.com -e PGADMIN_DEFAULT_PASSWORD=admin -p 15432:80 -d --network library-network dpage/pgadmin4

# inicializar o docker
systemctl --user start docker-desktop

#Verificando os containers existentes
docker ps -a

#Inicializando os containers existentes
docker start librarydb
docker start pgadmin4
